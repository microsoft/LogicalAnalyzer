91  public object? GetValueByPropertyString(object? targetObject, string pathToProperty)
92  {
93  if (pathToProperty is null || targetObject is null)
94  {
95  return null;
96  }
97  try
98  {
99  var pathPortions = pathToProperty.Split('.');
100 
101  // We first try to get the first value to get it started
102  var value = GetValueByPropertyOrFieldName(targetObject, pathPortions[0]);
103 
104  // For the rest of the path we walk each portion to get the next object
105  for (int pathPortionIndex = 1; pathPortionIndex < pathPortions.Length; pathPortionIndex++)
106  {
107  if (value == null) { break; }
108 
109  switch (value)
110  {
111  case Dictionary<string, string> stringDict:
112  if (stringDict.TryGetValue(pathPortions[pathPortionIndex], out string? stringValue))
113  {
114  value = stringValue;
115  }
116  else
117  {
118  value = null;
119  }
120  break;
121 
122  case List<string> stringList:
123  if (int.TryParse(pathPortions[pathPortionIndex], out int ArrayIndex) && stringList.Count > ArrayIndex)
124  {
125  value = stringList[ArrayIndex];
126  }
127  else
128  {
129  value = null;
130  }
131  break;
132 
133  default:
134  (bool Processed, object? Result)? res = null;
135  var found = false;
136  foreach(var del in CustomPropertyExtractionDelegates)
137  {
138  res = del?.Invoke(value, pathPortions[pathPortionIndex]);
139  if (res.HasValue && res.Value.Processed)
140  {
141  found = true;
142  value = res.Value.Result;
143  break;
144  }
145  }
146 
147  // If we couldn't do any custom parsing fall back to the default
148  if (!found)
149  {
150  value = GetValueByPropertyOrFieldName(value, pathPortions[pathPortionIndex]);
151  }
152  break;
153  }
154  }
155  return value;
156  }
157  catch (Exception e)
158  {
159  Log.Information("Fetching Field {0} failed from {1} ({2}:{3})", pathToProperty, targetObject.GetType(), e.GetType(), e.Message);
160  }
161  return null;
162  }

184  public string[] GetTags(IEnumerable<Rule> rules, object? state1 = null, object? state2 = null)
185  {
186  var tags = new ConcurrentDictionary<string, byte>();
187 
188  Parallel.ForEach(rules, rule =>
189  {
190  if (!rule.Tags.All(x => tags.Keys.Any(y => y == x)) && Applies(rule, state1, state2))
191  {
192  foreach (var tag in rule.Tags)
193  {
194  tags.TryAdd(tag, 0);
195  }
196  }
197  });
198 
199  return tags.Keys.ToArray();

210  {

213  Parallel.ForEach(rules, rule =>

231  public bool Applies(Rule rule, object? state1 = null, object? state2 = null)
232  {
233  if (rule != null)
234  {
235  var sample = state1 is null ? state2 : state1;
236 
237  // Does the name of this class match the Target in the rule?
238  // Or has no target been specified (match all)
239  if (rule.Target is null || (sample?.GetType().Name.Equals(rule.Target, StringComparison.InvariantCultureIgnoreCase) ?? true))
240  {
241  // If the expression is null the default is that all clauses must be true
242  // If we have no clauses .All will still match
243  if (rule.Expression is null)
244  {
245  if (rule.Clauses.All(x => AnalyzeClause(x, state1, state2)))
246  {
247  return true;
248  }
249  }
250  // Otherwise we evaluate the expression
251  else
252  {
253  if (Evaluate(rule.Expression.Split(' '), rule.Clauses, state1, state2))
254  {
255  return true;
256  }
257  }
258  }
259 
260  return false;
261  }
262  else
263  {
264  throw new NullReferenceException();
265  }
266  }

280  public IEnumerable<Violation> EnumerateRuleIssues(IEnumerable<Rule> rules)
281  {
282  if (!Strings.IsLoaded)
283  {
284  Strings.Setup();
285  }
286  foreach (Rule rule in rules ?? Array.Empty<Rule>())
287  {
288  var clauseLabels = rule.Clauses.GroupBy(x => x.Label);
289 
290  // If clauses have duplicate names
291  var duplicateClauses = clauseLabels.Where(x => x.Key != null && x.Count() > 1);
292  foreach (var duplicateClause in duplicateClauses)
293  {
294  yield return new Violation(string.Format(Strings.Get("Err_ClauseDuplicateName"), rule.Name, duplicateClause.Key ?? string.Empty), rule, duplicateClause.AsEnumerable().ToArray());
295  }
296 
297  // If clause label contains illegal characters
298  foreach (var clause in rule.Clauses)
299  {
300  if (clause.Label is string label)
301  {
302  if (label.Contains(" ") || label.Contains("(") || label.Contains(")"))
303  {
304  yield return new Violation(string.Format(Strings.Get("Err_ClauseInvalidLabel"), rule.Name, label), rule, clause);
305  }
306  }
307  switch (clause.Operation)
308  {
309  case OPERATION.EQ:
310  case OPERATION.NEQ:
311  if ((clause.Data?.Count == null || clause.Data?.Count == 0))
312  {
313  yield return new Violation(string.Format(Strings.Get("Err_ClauseNoData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
314  }
315  if (clause.DictData != null || clause.DictData?.Count > 0)
316  {
317  yield return new Violation(string.Format(Strings.Get("Err_ClauseDictDataUnexpected"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), clause.Operation.ToString()), rule, clause);
318  }
319  break;
320 
321  case OPERATION.CONTAINS:
322  case OPERATION.CONTAINS_ANY:
323  if ((clause.Data?.Count == null || clause.Data?.Count == 0) && (clause.DictData?.Count == null || clause.DictData?.Count == 0))
324  {
325  yield return new Violation(string.Format(Strings.Get("Err_ClauseNoDataOrDictData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
326  }
327  if ((clause.Data is List<string> list && list.Count > 0) && (clause.DictData is List<KeyValuePair<string, string>> dictList && dictList.Count > 0))
328  {
329  yield return new Violation(string.Format(Strings.Get("Err_ClauseBothDataDictData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
330  }
331  break;
332 
333  case OPERATION.ENDS_WITH:
334  case OPERATION.STARTS_WITH:
335  if (clause.Data?.Count == null || clause.Data?.Count == 0)
336  {
337  yield return new Violation(string.Format(Strings.Get("Err_ClauseNoData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
338  }
339  if (clause.DictData != null || clause.DictData?.Count > 0)
340  {
341  yield return new Violation(string.Format(Strings.Get("Err_ClauseDictDataUnexpected"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), clause.Operation.ToString()), rule, clause);
342  }
343  break;
344 
345  case OPERATION.GT:
346  case OPERATION.LT:
347  if (clause.Data?.Count == null || clause.Data is List<string> clauseList && (clauseList.Count != 1 || !int.TryParse(clause.Data.First(), out int _)))
348  {
349  yield return new Violation(string.Format(Strings.Get("Err_ClauseExpectedInt"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
350  }
351  if (clause.DictData != null || clause.DictData?.Count > 0)
352  {
353  yield return new Violation(string.Format(Strings.Get("Err_ClauseDictDataUnexpected"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), clause.Operation.ToString()), rule, clause);
354  }
355  break;
356 
357  case OPERATION.REGEX:
358  if (clause.Data?.Count == null || clause.Data?.Count == 0)
359  {
360  yield return new Violation(string.Format(Strings.Get("Err_ClauseNoData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
361  }
362  else if (clause.Data is List<string> regexList)
363  {
364  foreach (var regex in regexList)
365  {
366  if (!Helpers.IsValidRegex(regex))
367  {
368  yield return new Violation(string.Format(Strings.Get("Err_ClauseInvalidRegex"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), regex), rule, clause);
369  }
370  }
371  }
372  if (clause.DictData != null || clause.DictData?.Count > 0)
373  {
374  yield return new Violation(string.Format(Strings.Get("Err_ClauseDictDataUnexpected"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), clause.Operation.ToString()), rule, clause);
375  }
376  break;
377 
378  case OPERATION.IS_NULL:
379  case OPERATION.IS_TRUE:
380  case OPERATION.IS_EXPIRED:
381  case OPERATION.WAS_MODIFIED:
382  if (!(clause.Data?.Count == null || clause.Data?.Count == 0))
383  {
384  yield return new Violation(string.Format(Strings.Get("Err_ClauseRedundantData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
385  }
386  else if (!(clause.DictData?.Count == null || clause.DictData?.Count == 0))
387  {
388  yield return new Violation(string.Format(Strings.Get("Err_ClauseRedundantDictData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
389  }
390  break;
391 
392  case OPERATION.IS_BEFORE:
393  case OPERATION.IS_AFTER:
394  if (clause.Data?.Count == null || clause.Data is List<string> clauseList2 && (clauseList2.Count != 1 || !DateTime.TryParse(clause.Data.First(), out DateTime _)))
395  {
396  yield return new Violation(string.Format(Strings.Get("Err_ClauseExpectedDateTime"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
397  }
398  if (clause.DictData != null || clause.DictData?.Count > 0)
399  {
400  yield return new Violation(string.Format(Strings.Get("Err_ClauseDictDataUnexpected"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), clause.Operation.ToString()), rule, clause);
401  }
402  break;
403 
404  case OPERATION.CONTAINS_KEY:
405  if (clause.DictData != null)
406  {
407  yield return new Violation(string.Format(Strings.Get("Err_ClauseUnexpectedDictData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
408  }
409  if (clause.Data == null || clause.Data?.Count == 0)
410  {
411  yield return new Violation(string.Format(Strings.Get("Err_ClauseMissingListData"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
412  }
413  break;
414 
415  case OPERATION.CUSTOM:
416  if (clause.CustomOperation == null)
417  {
418  yield return new Violation(string.Format(Strings.Get("Err_ClauseMissingCustomOperation"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
419  }
420  else
421  {
422  bool covered = false;
423  foreach (var del in CustomOperationValidationDelegates)
424  {
425  var res = del?.Invoke(rule, clause) ?? (false, new List<Violation>());
426  if (res.Applies)
427  {
428  covered = true;
429  foreach (var violation in res.FoundViolations)
430  {
431  yield return violation;
432  }
433  }
434  }
435  if (covered == false)
436  {
437  yield return new Violation(string.Format(Strings.Get("Err_ClauseMissingValidationForOperation"), clause.CustomOperation, rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture)), rule, clause);
438  }
439  }
440  break;
441 
442  case OPERATION.DOES_NOT_CONTAIN:
443  case OPERATION.DOES_NOT_CONTAIN_ALL:
444  default:
445  yield return new Violation(string.Format(Strings.Get("Err_ClauseUnsuppportedOperator"), rule.Name, clause.Label ?? rule.Clauses.IndexOf(clause).ToString(CultureInfo.InvariantCulture), clause.Operation.ToString()), rule, clause);
446  break;
447  }
448  }
449 
450  var foundLabels = new List<string>();
451 
452  if (rule.Expression is string expression)
453  {
454  // Are parenthesis balanced Are spaces correct Are all variables defined by
455  // clauses? Are variables and operators alternating?
456  var splits = expression.Split(' ');
457  int foundStarts = 0;
458  int foundEnds = 0;
459  bool expectingOperator = false;
460  for (int i = 0; i < splits.Length; i++)
461  {
462  foundStarts += splits[i].Count(x => x.Equals('('));
463  foundEnds += splits[i].Count(x => x.Equals(')'));
464  if (foundEnds > foundStarts)
465  {
466  yield return new Violation(string.Format(Strings.Get("Err_ClauseUnbalancedParentheses"), expression, rule.Name), rule);
467  }
468  // Variable
469  if (!expectingOperator)
470  {
471  var lastOpen = -1;
472  var lastClose = -1;
473 
474  for (int j = 0; j < splits[i].Length; j++)
475  {
476  // Check that the parenthesis are balanced
477  if (splits[i][j] == '(')
478  {
479  // If we've seen a ) this is now invalid
480  if (lastClose != -1)
481  {
482  yield return new Violation(string.Format(Strings.Get("Err_ClauseParenthesisInLabel"), expression, rule.Name, splits[i]), rule);
483  }
484  // If there were any characters between open parenthesis
485  if (j - lastOpen != 1)
486  {
487  yield return new Violation(string.Format(Strings.Get("Err_ClauseCharactersBetweenOpenParentheses"), expression, rule.Name, splits[i]), rule);
488  }
489  // If there was a random parenthesis not starting the variable
490  else if (j > 0)
491  {
492  yield return new Violation(string.Format(Strings.Get("Err_ClauseCharactersBeforeOpenParentheses"), expression, rule.Name, splits[i]), rule);
493  }
494  lastOpen = j;
495  }
496  else if (splits[i][j] == ')')
497  {
498  // If we've seen a close before update last
499  if (lastClose != -1 && j - lastClose != 1)
500  {
501  yield return new Violation(string.Format(Strings.Get("Err_ClauseCharactersBetweenClosedParentheses"), expression, rule.Name, splits[i]), rule);
502  }
503  lastClose = j;
504  }
505  else
506  {
507  // If we've set a close this is invalid because we can't have
508  // other characters after it
509  if (lastClose != -1)
510  {
511  yield return new Violation(string.Format(Strings.Get("Err_ClauseCharactersAfterClosedParentheses"), expression, rule.Name, splits[i]), rule);
512  }
513  }
514  }
515 
516  var variable = splits[i].Replace("(", "").Replace(")", "");
517 
518  if (variable == "NOT")
519  {
520  if (splits[i].Contains(")"))
521  {
522  yield return new Violation(string.Format(Strings.Get("Err_ClauseCloseParenthesesInNot"), expression, rule.Name, splits[i]), rule);
523  }
524  }
525  else
526  {
527  foundLabels.Add(variable);
528  if (string.IsNullOrWhiteSpace(variable) || !rule.Clauses.Any(x => x.Label == variable))
529  {
530  yield return new Violation(string.Format(Strings.Get("Err_ClauseUndefinedLabel"), expression, rule.Name, splits[i].Replace("(", "").Replace(")", "")), rule);
531  }
532  expectingOperator = true;
533  }
534  }
535  //Operator
536  else
537  {
538  // If we can't enum parse the operator
539  if (!Enum.TryParse<BOOL_OPERATOR>(splits[i], out BOOL_OPERATOR op))
540  {
541  yield return new Violation(string.Format(Strings.Get("Err_ClauseInvalidOperator"), expression, rule.Name, splits[i]), rule);
542  }
543  // We don't allow NOT operators to modify other Operators, so we can't
544  // allow NOT here
545  else
546  {
547  if (op is BOOL_OPERATOR boolOp && boolOp == BOOL_OPERATOR.NOT)
548  {
549  yield return new Violation(string.Format(Strings.Get("Err_ClauseInvalidNotOperator"), expression, rule.Name), rule);
550  }
551  }
552  expectingOperator = false;
553  }
554  }
555 
556  // We should always end on expecting an operator (having gotten a variable)
557  if (!expectingOperator)
558  {
559  yield return new Violation(string.Format(Strings.Get("Err_ClauseEndsWithOperator"), expression, rule.Name), rule);
560  }
561  }
562 
563  // Were all the labels declared in clauses used?
564  foreach (var label in rule.Clauses.Select(x => x.Label))
565  {
566  if (label is string)
567  {
568  if (!foundLabels.Contains(label))
569  {
570  yield return new Violation(string.Format(Strings.Get("Err_ClauseUnusedLabel"), label, rule.Name), rule);
571  }
572  }
573  }
574 
575  var justTheLabels = clauseLabels.Select(x => x.Key);
576  // If any clause has a label they all must have labels
577  if (justTheLabels.Any(x => x is string) && justTheLabels.Any(x => x is null))
578  {
579  yield return new Violation(string.Format(Strings.Get("Err_ClauseMissingLabels"), rule.Name), rule);
580  }
581  // If the clause has an expression it may not have any null labels
582  if (rule.Expression != null && justTheLabels.Any(x => x is null))
583  {
584  yield return new Violation(string.Format(Strings.Get("Err_ClauseExpressionButMissingLabels"), rule.Name), rule);
585  }
586  }
587  }

596  public bool AnalyzeClause(Clause clause, object? state1 = null, object? state2 = null)
597  {
598  if (clause == null)
599  {
600  return false;
601  }
602  try
603  {
604  if (clause.Field is string)
605  {
606  state2 = GetValueByPropertyString(state2, clause.Field);
607  state1 = GetValueByPropertyString(state1, clause.Field);
608  }
609 
610  var typeHolder = state1 is null ? state2 : state1;
611 
612  (var stateOneList, var stateOneDict) = ObjectToValues(state1);
613  (var stateTwoList, var stateTwoDict) = ObjectToValues(state2);
614 
615  var valsToCheck = stateOneList.Union(stateTwoList);
616  var dictToCheck = stateOneDict.Union(stateTwoDict);
617 
618  switch (clause.Operation)
619  {
620  case OPERATION.EQ:
621  if (clause.Data is List<string> EqualsData)
622  {
623  if (EqualsData.Intersect(valsToCheck).Any())
624  {
625  return true;
626  }
627  }
628  return false;
629 
630  case OPERATION.NEQ:
631  if (clause.Data is List<string> NotEqualsData)
632  {
633  if (!NotEqualsData.Intersect(valsToCheck).Any())
634  {
635  return true;
636  }
637  }
638  return false;
639 
640  // If *every* entry of the clause data is matched
641  case OPERATION.CONTAINS:
642  if (dictToCheck.Any())
643  {
644  if (clause.DictData is List<KeyValuePair<string, string>> ContainsData)
645  {
646  if (ContainsData.All(y => dictToCheck.Any((x) => x.Key == y.Key && x.Value == y.Value)))
647  {
648  return true;
649  }
650  }
651  }
652  else if (valsToCheck.Any())
653  {
654  if (clause.Data is List<string> ContainsDataList)
655  {
656  // If we are dealing with an array on the object side
657  if (typeHolder is List<string>)
658  {
659  if (ContainsDataList.All(x => valsToCheck.Contains(x)))
660  {
661  return true;
662  }
663  }
664  // If we are dealing with a single string we do a .Contains instead
665  else if (typeHolder is string)
666  {
667  if (clause.Data.All(x => valsToCheck.First()?.Contains(x) ?? false))
668  {
669  return true;
670  }
671  }
672 
673  else if (typeHolder?.GetType().IsDefined(typeof(FlagsAttribute), false) is true)
674  {
675  var enums = new List<Enum>();
676  foreach (var datum in clause.Data ?? new List<string>())
677  {
678  if (Enum.TryParse(typeHolder.GetType(), datum, out object result))
679  {
680  if (!(state1 is Enum enum1 && enum1.HasFlag((Enum)result)))
681  {
682  return false;
683  }
684  }
685  else
686  {
687  return false;
688  }
689  }
690  return true;
691  }
692  }
693  }
694  return false;
695 
696  // If *any* entry of the clause data is matched
697  case OPERATION.CONTAINS_ANY:
698  if (dictToCheck.Any())
699  {
700  if (clause.DictData is List<KeyValuePair<string, string>> ContainsData)
701  {
702  foreach (KeyValuePair<string, string> value in ContainsData)
703  {
704  if (dictToCheck.Any(x => x.Key == value.Key && x.Value == value.Value))
705  {
706  return true;
707  }
708  }
709  }
710  }
711  else if (valsToCheck.Any())
712  {
713  if (clause.Data is List<string> ContainsDataList)
714  {
715  if (typeHolder is List<string>)
716  {
717  if (ContainsDataList.Any(x => valsToCheck.Contains(x)))
718  {
719  return true;
720  }
721  }
722  // If we are dealing with a single string we do a .Contains instead
723  else if (typeHolder is string)
724  {
725  if (clause.Data.Any(x => valsToCheck.First()?.Contains(x) ?? false))
726  {
727  return true;
728  }
729  }
730 
731  else if (typeHolder?.GetType().IsDefined(typeof(FlagsAttribute), false) is true)
732  {
733  var enums = new List<Enum>();
734  foreach (var datum in clause.Data ?? new List<string>())
735  {
736  if (Enum.TryParse(typeHolder.GetType(), datum, out object result))
737  {
738  if (state1 is Enum enum1 && enum1.HasFlag((Enum)result))
739  {
740  return true;
741  }
742  }
743  }
744  }
745  }

1069  private bool Evaluate(string[] splits, List<Clause> Clauses, object? state1, object? state2)
1070  {
1071  bool current = false;
1072 
1073  var invertNextStatement = false;
1074  var operatorExpected = false;
1075 
1076  BOOL_OPERATOR Operator = BOOL_OPERATOR.OR;
1077 
1078  var updated_i = 0;
1079 
1080  for (int i = 0; i < splits.Length; i = updated_i)
1081  {
1082  if (operatorExpected)
1083  {
1084  Operator = (BOOL_OPERATOR)Enum.Parse(typeof(BOOL_OPERATOR), splits[i]);
1085  operatorExpected = false;
1086  updated_i = i + 1;
1087  }
1088  else
1089  {
1090  if (splits[i].StartsWith("("))
1091  {
1092  //Get the substring closing this paren
1093  var matchingParen = FindMatchingParen(splits, i);
1094 
1095  // First remove the parenthesis from the beginning and end
1096  splits[i] = splits[i][1..];
1097  splits[matchingParen] = splits[matchingParen][0..^1];
1098 
1099  var shortcut = TryShortcut(current, Operator);
1100 
1101  if (shortcut.CanShortcut)
1102  {
1103  current = shortcut.Value;
1104  }
1105  else
1106  {
1107  // Recursively evaluate the contents of the parentheses
1108  var next = Evaluate(splits[i..(matchingParen + 1)], Clauses, state1, state2);
1109 
1110  next = invertNextStatement ? !next : next;
1111 
1112  current = Operate(Operator, current, next);
1113  }
1114 
1115  updated_i = matchingParen + 1;
1116  invertNextStatement = false;
1117  operatorExpected = true;
1118  }
1119  else
1120  {
1121  if (splits[i].Equals(BOOL_OPERATOR.NOT.ToString()))
1122  {
1123  invertNextStatement = !invertNextStatement;
1124  operatorExpected = false;
1125  }
1126  else
1127  {
1128  // Ensure we have exactly 1 matching clause defined
1129  var res = Clauses.Where(x => x.Label == splits[i].Replace("(", "").Replace(")", ""));
1130  if (!(res.Count() == 1))
1131  {
1132  return false;
1133  }
1134 
1135  var clause = res.First();
1136 
1137  var shortcut = TryShortcut(current, Operator);
1138 
1139  if (shortcut.CanShortcut)
1140  {
1141  current = shortcut.Value;
1142  }
1143  else
1144  {
1145  bool next;
1146 
1147  next = AnalyzeClause(res.First(), state1, state2);
1148 
1149  next = invertNextStatement ? !next : next;
1150 
1151  current = Operate(Operator, current, next);
1152  }
1153 
1154  invertNextStatement = false;
1155  operatorExpected = true;
1156  }
1157  updated_i = i + 1;
1158  }
1159  }
1160  }
1161  return current;

1170  public static (bool CanShortcut, bool Value) TryShortcut(bool current, BOOL_OPERATOR operation)
1171  {
1172  // If either argument of an AND statement is false, or either argument of a
1173  // NOR statement is true, the result is always false and we can optimize
1174  // away evaluation of next
1175  if ((operation == BOOL_OPERATOR.AND && current == false) ||
1176  (operation == BOOL_OPERATOR.NOR && current == true))
1177  {
1178  return (true, false);
1179  }
1180  // If either argument of an NAND statement is false, or either argument of
1181  // an OR statement is true, the result is always true and we can optimize
1182  // away evaluation of next
1183  if ((operation == BOOL_OPERATOR.OR && current == true) ||
1184  (operation == BOOL_OPERATOR.NAND && current == false))
1185  {
1186  return (true, true);
1187  }
1188  return (false, false);
